Getting touch points of canvas

element.getBoundingClientRect()returns an object containing all the values of 'element'

. alert(canvas.getBoundingClientRect().width);//300 alert(canvas.getBoundingClientRect().height);//300 alert(canvas.getBoundingClientRect().left);//8 alert(canvas.getBoundingClientRect().top);//433.4375(from the top of the body to the top of the box 
canvas.addEventListener("mousemove",function(e){
  alert(e.clientX)//returns the X coordinate(number) of the clicked position from the left side of body
  alert(e.clientY)//returns the Y coordinate(number) of the clicked position from the Top(beginning)  of the body(webpage)
  
});
